 standard input , standardoutput
  
        standard input : همون کیبورد ما است
        standard output : همون اسکرین ما است

        نکته : 
               ما میتوانیم این دوتارو تغییر بدیم
               و به این کار ریدایرکشن میگویند

         cat <fileName> : 
                          خب مبخواهبم محتوای فایل را ببینیم
                         درواقع زمانی که اینکار را اناجم میدهیم
                         خروجی در استاندارد اوت پوت ما نمایش داده میشود

       cat file1.txt > file2.txt : 
                                    بااین روش میتوانیم خروجی را به جای اسکرین
                                    توی فایل دیگر نمایش دهیم ودرواقع برای ما
                                    این فایل را کپی کرده

  خب حالا میخواهیم برسی کنیم چگونه میتوانیم یک رشته را در یک فایل جستجو کنیم

     grep:() : 

              g → global
              re → regular expression (عبارت منظم)
              p → print
              ----- - - - 
              میتوانیم تنظیم کنیم به حروف بزرگ و کوچک حساس باشد
              یا حساس نباشد : 
                             1- اگربخواهیم حساس باشد : 
                                                       grep hello file.txt => output => hello (اگر کلمه ای باشد با رنگ قرمز نمایش میدهد )
                                                       دراین حالت اگر سلام را با حرف بزرگ جستجو میکردیم چیزی به ما 
                                                       برنمیگرداند

                             2- حالا نمیخواهیم به حروف حساس باشد : 
                                                        grep -i Hello file1.txt => output => Hello my friend    hello      
                                                        این دفعه چون حساسیت را غیر فعال کردیم خروجی هردو سلام
                                                        با حروف بزرگ و کوچک را اورد                   
                                                        

                             3- حالا اگر بخواهیم در دویاچندفایل دنبال کلمه ای بگردیم : 
                                                             
                                                         grep -i hello file1.txt file2.txt

                              4- خب حالا میخواهیم با الگویی فایل رو پیدا کنیم و داخلش جستجو کنیم : 

                                                          grep -i Hello file* :  اون فایل هایی که با این حرف شروع شده
                                                                                  و بعد هرچیزی میخواهند داشته باشد   

                                                          grep -i Hello *.txt : اون فایل هایی که با این پسوند
                                                                                 تمام میشن
      ------- -- - - - - - - - - 

                              5- خب میتوانیم سرچ را در دایرکتوری های دیگر انجام دهیم : 
                                                                                      -r بااین فلگ 
                                                          grep -i -r hello . :
                                                                              خب ما میخواستیم در این دایرکتوری
                                                                              برای ما سرچ را انجام دهد ودرهر فایل
                                                                              انجام میدهد

                                                          grep -i -r hello /etc : 
                                                                                 میتوانیم دایرکتوری را انتخاب کنبم

       جستجوی فایل و دایرکتوری : 
                                 
                                 find : 
                                        مبتوانیم فایل ها و دایرکتوری هارا جستجو کنیم
                                        زمانی که این دستور رو خالی اجرا کنی توی روتی که هستی
                                        تمام فایل ها و فولدر هارو بهم برمیگردونه
                                        نه تنها محتوای دایرکتوری که توش هست را میخواند 
                                        وحتی اگر داخل اون دایرکتوری ها بازهم دایرکتوری
                                        باشه اوناروهم میخونه 

                                        فایل های مخفی رو بهمون نشون میده
                                        فایل های مخفی رو نشون نمیدهls درصورتی که 
                                        
 به این صورت نیز میتوانیم فایل های هیدن رامشاهده کنیم : 
                                                        ls -a


-------- - - - - - - - - - - -
      find /etc : 
                 ودرون ان جستجو رابرای ما انجام میدهد find میتوانیم ادرس یک فولدر را به 

      find -type d : 
                     فقط برای ما دایرکتوری هارانمایش میدهد

      find -type f : 
                     میتوانیم فایل هارا جستجو کنیم                          

   ------------ - - - - - - - - - - - 

 میتوانیم براساس نام فایل ها یا دایرکتوری ها فیلتر نیز انجام دهیم : 

      find -type f -name "f*" : 
                                 تمام فایل هایی که با اف شروع میشود و دیگر مهم نیست
                                 حروف بعدیش چیه
       
      find -type f -iname "f*" :
                                  اگر بخواهیم به حروف بزرگ و کوچک حساس نباشد

 زنجیر کردن کامند ها  : 
                         درلینوکس ما میتوانیم کامند هارا با یکدیگر زنجیر کنیم
                         یعنی میتوانیم به یکباره ان هارا اجرا کنیم

         مثال : 

   mkdir  test ; cd test ; echo done  : 

                                  میخوام به یک دایرکتوری وارد بشم و یک گیغامی رو چاپ کنم : 
  
                                                 
   mkdir  test && cd test && echo done : 

                                    اگر بخواهیم در هنگام بروز خطلا در یکی از
                                    نصب ها بقیه دستورات هم متوقف بشه                      

   mkdir  test || cd test || echo done : 

                                   اگر کامندی اجرا شود به سراغ کامند بعدی نمیرد
                                   اگر کامندی به خطلا بخورد به سراغ کامند های بعدی
                                   میرود                                 

    ls /bin | less | ls /bin | head  |  ls /bin | tail : 
                      bin  در این مرحله داریم خروجی فولدر      
                      پاس میدهیم less  رو به ورودی  
                      توسط | مامیتوانیم روی موارد دیگر 
                      نیز از قابلیت پایپ استفاده کنیم
